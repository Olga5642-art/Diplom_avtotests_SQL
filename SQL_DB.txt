Задание_1_БД:
SELECT c.login, COUNT (o.track) FROM "Couriers" AS c LEFT OUTER JOIN "Orders" AS o ON c.id=o."courierId" WHERE o."inDelivery"='t' GROUP BY c.login;

SELECT track, CASE WHEN finished='t' THEN '2' WHEN cancelled='t' THEN '-1' WHEN "inDelivery"='t' THEN '1' ELSE '0' END FROM "Orders";
